NOTE: Namespaced mode is enabled (watchNamespace={{ .Values.controller.watchNamespace }}).

In namespaced mode the controller skips webhook certificate generation, CRD
caBundle patching, and the webhook server. Conversion webhooks are required when
etcd still contains v1alpha1 objects and the API server needs to translate them
to v1beta1. Without a serving conversion webhook, requests that require
conversion will fail. Migrate all stored objects to v1beta1 before using
namespaced mode, or keep a cluster-scoped conversion webhook available.

apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: agent-sandbox-controller-leader-election
  namespace: {{ $leaderElectionNamespace }}
rules:
- apiGroups:
  - ""
  resources:
  - events
  verbs:
  - create
  - patch
- apiGroups:
  - coordination.k8s.io
  resources:
  - leases
  verbs:
  - create
  - get
  - list
  - patch
  - update
  - watch
---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: agent-sandbox-controller-leader-election
  namespace: {{ $leaderElectionNamespace }}
subjects:
- kind: ServiceAccount
  name: agent-sandbox-controller
  namespace: {{ $controllerNamespace }}
roleRef:
  kind: Role
  name: agent-sandbox-controller-leader-election
  apiGroup: rbac.authorization.k8s.io
